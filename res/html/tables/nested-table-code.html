HTML


  <h2>Nested Table</h2>
  <p>An example of how to create a table inside a table:</p>
  
  <table>
    <tr>
      <td>Cell</td>
      <td>Nested Table
        <table>
          <tr>
            <td>Cell 1</td>
          </tr>
          <tr>
            <td>Cell 2</td>
          </tr>
          <tr>
            <td>Cell 3</td>
          </tr>
          <tr>
            <td>Cell 4</td>
          </tr>
        </table>
      </td>
    </tr>
  </table>

CSS


  table {
    width: 100%;
    border: 1px solid #ddd;
  }
  
  td {
    padding: 16px;
  }
  
  tr:nth-child(even) {
    background-color: #eee;
  }
