  const points1 = [40, 100, 1, 5, 25, 10];
  document.getElementById("demo1").innerHTML = points1;  
  
  points1.sort(function(a, b){return a - b});
  document.getElementById("demo2").innerHTML = points1;
  
  const points2 = [40, 100, 1, 5, 25, 10];
  document.getElementById("demo3").innerHTML = points2;  
  
  points2.sort(function(a, b){return b - a});
  document.getElementById("demo4").innerHTML = points2;
